def roc(df):
    return pd.DataFrame([[roc_auc_score(df.y_hypo, df.probas_lr_two),
                          #accuracy_score(df.y_hypo_3, df.probas_lr_two), 
                          df.shape[0],
                          len(set(df.ID_y)), df.y_hypo.mean()]], columns=['roc', 'bouts',
                                                      'participants', 'ratio hypos'])
